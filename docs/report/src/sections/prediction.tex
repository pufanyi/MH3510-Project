\begin{equation}
    \hat{y} = \exp{\sqrt[4]{\hat{\beta}_0 +\hat{\beta}_1 \sqrt{x_1} + \hat{\beta}_2 x_2 + \hat{\beta}_4 (2 - x_4)}}
\end{equation}

Substituting by
\begin{equation}
\begin{cases}
x1 = 50000\\
x2 = 3\\
x3 = 60\\
x4 = 2\\
\end{cases}
\end{equation}

We can get
\begin{equation}\label{eq:prediction}
\hat{y}\approx 27379.73
\end{equation}
